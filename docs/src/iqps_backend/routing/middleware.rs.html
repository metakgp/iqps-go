14/// Verifies the JWT and authenticates a user. If the JWT is invalid, the user is sent an unauthorized status code. If the JWT is valid, the authentication is added to the state.
15pub async fn verify_jwt_middleware(
16    State(state): State<RouterState>,
17    headers: HeaderMap,
18    mut request: Request,
19    next: Next,
20) -> Result<Response, AppError> {
21    if let Some(auth_header) = headers.get("Authorization") {
22        if let Some(jwt) = auth_header.to_str()?.strip_prefix("Bearer ") {
23            let auth = auth::verify_token(jwt, &state.env_vars).await;
24
25            if let Ok(auth) = auth {
26                // If auth is fine, add it to the request extensions
27                request.extensions_mut().insert(auth);
28                Ok(next.run(request).await)
29            } else {
30                Ok(BackendResponse::<()>::error(
31                    "Authorization token invalid.".into(),
32                    StatusCode::UNAUTHORIZED,
33                )
34                .into_response())
35            }
36        } else {
37            Ok(BackendResponse::<()>::error(
38                "Authorization header format invalid.".into(),
39                StatusCode::UNAUTHORIZED,
40            )
41            .into_response())
42        }
43    } else {
44        Ok(BackendResponse::<()>::error(
45            "Authorization header missing.".into(),
46            StatusCode::UNAUTHORIZED,
47        )
48        .into_response())
49    }
50}
